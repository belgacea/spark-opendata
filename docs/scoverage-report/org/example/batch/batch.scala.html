33     /**
34      * Street column fallback expression
35      *
36      * @param column1 Street column 1
37      * @param column2 Street column 2
38      * @return Street column
39      */
40     def getStreetColExpr(column1: Column, column2: Column): Column = {
41       when(column1.isNotNull and !(column1 <=> lit("")),
42         column1)
43         .otherwise(
44           when(column2.isNotNull and !(column2 <=> lit("")),
45             column2)
46             // .otherwise(lit("Unknown"))
47             .otherwise(
48               when(col(address1Col).isNotNull and !(col(address1Col) <=> lit("")),
49                 col(address1Col))
50                 .otherwise(
51                   lit("Unknown"))
52             )
53         )
54     }

56     /**
57      * Get max visitor by street (filtered by accommodation)
58      *
59      * @group aggregation
60      * @param dataFrame Input dataframe
61      * @param accommodation Accommodation filter value
62      */
63     def maxVisitorByStreet(dataFrame: DataFrame, accommodation: Boolean = true): DataFrame = {
64       val accommodationValue = if (accommodation) "0" else "N"
65       dataFrame
66         .filter(col(accommodationCol) <=> lit(accommodationValue)) // Filter by accommodation column
67         .withColumn(streetCol1, streetCol1Epxr)
68         .withColumn(streetCol2, streetCol2Epxr)
69         .withColumn(street, streetColEpxr) // Consolidate street column
70         .groupBy(street)
71         .agg(sum(col("nb_visiteurs_max")).as("max_visitors")) // Process max visitor by street
72     }
